18 int& ntime,
19 bool& read_flag,
20 bool& resize_flag,
21 amrex::Real& wtime,
22 amrex::Real& t_last,
23 const int new_ntime,
24 const amrex::Real wtinit,
25 const amrex::Real wdt,
26 const amrex::Real time)
27{
28 // Flag to indicate that data must be read twice for interpolation
29 int double_data = 0;
30 // Check if time indicates reading must take place
31 if (new_ntime != ntime) {
32 // New data is needed - reading should happen
33 read_flag = true;
34 // If time index has proceeded more than 1 step
35 if (new_ntime > ntime + 1) {
36 // Double reading is necessary
37 double_data = 1;
38 }
39 // Time index for reading
40 ntime = new_ntime;
41 // Sim time to go with recorded data
42 wtime = new_ntime * wdt + wtinit;
43 // If double reading is deemed necessary, check for convenience
44 if (double_data == 1 && std::abs(wtime - time) <= 1e-10) {
45 // Reading can be done just once, w2a fields replace ow fields
46 double_data = 2;
47 }
48 }
49 // Check if reading must take place for other reasons
50 if (t_last < -1e-10) {
51 // Signifies initialization from scratch without waves or a restart
52 read_flag = true;
53 // Resizing needs to happen for the first time
54 resize_flag = true;
55
56 // Confirm that new time is not coincident with modes time step
57 if (std::abs(wtime - time) > 1e-10) {
58 // Data must be read before and after wtime
59 double_data = 1;
60 } else {
61 // Data need only be read once, w2a fields will replace ow fields
62 double_data = 2;
63 }
64
65 } else if (std::abs(t_last) < 1e-10) {
66 // Signifies initialization with waves
67 read_flag = true;
68 // Resizing needs to happen for the first time
69 resize_flag = true;
70
71 // levelset and velocity fields are up-to-date at t=0
72 // interpolation is ready to go
73 }
74 // Record latest time as 'last' for next timestep
75 t_last = time;
76 // Return flag regarding double reading
77 return double_data;
78}

80void postprocess_velocity_mfab_liquid(
81 amrex::MultiFab& vel_mfab,
82 amrex::MultiFab& lvs_mfab,
83 const amrex::GpuArray<amrex::Real, AMREX_SPACEDIM> dx)
84{
85 auto vel = vel_mfab.arrays();
86 const auto phi = lvs_mfab.const_arrays();
87 amrex::ParallelFor(
88 vel_mfab, amrex::IntVect(3),
89 [=] AMREX_GPU_DEVICE(int nbx, int i, int j, int k) noexcept {
90 // Set velocity to zero if no liquid present
91 const amrex::Real cell_length_2D =
92 std::sqrt(dx[0] * dx[0] + dx[2] * dx[2]);
93 if (phi[nbx](i, j, k) + cell_length_2D < 0) {
94 vel[nbx](i, j, k, 0) = 0.0;
95 vel[nbx](i, j, k, 1) = 0.0;
96 vel[nbx](i, j, k, 2) = 0.0;
97 }
98 });
99 amrex::Gpu::streamSynchronize();
100}

511 wdata.r_rmodes, wdata.plan_vector, wdata.r_eta_mptr,
512 wdata.sp_eta_vec);
513 // Mesh is not yet created, so get data at every height
514 const auto n_hts = wdata.hvec.size();
515 wdata.sp_u_vec.resize(

531 }

541 w2a_levelset.set_default_fillpatch_bc(data.sim().time());
542 w2a_velocity.set_default_fillpatch_bc(data.sim().time());
543#endif

547template <>
548struct InitDataOp<W2AWaves>
549{
550 void
551 // cppcheck-suppress constParameterReference
552 operator()(
553 W2AWaves::DataType &
554 data, int level, const amrex::Geometry & geom, bool multiphase_mode)
555 {
556
557#ifdef AMR_WIND_USE_W2A
558 auto& wdata = data.meta();
559
560 auto& sim = data.sim();
561
562 // Fill ow fields, then populate flow fields according to setup
563 auto& ow_levelset = sim.repo().get_field("ow_levelset");
564 auto& ow_velocity = sim.repo().get_field("ow_velocity");
565
566 auto& velocity = sim.repo().get_field("velocity");
567 // Set w2a fields to default values to prep for updates
568 auto& w2a_levelset = sim.repo().get_field("w2a_levelset");
569 auto& w2a_velocity = sim.repo().get_field("w2a_velocity");
570
571 Field* levelset{nullptr};
572 if (multiphase_mode) {
573 levelset = &sim.repo().get_field("levelset");
574 }
575
576 const auto& problo = geom.ProbLoArray();
577 const auto& probhi = geom.ProbHiArray();
578 const auto& dx = geom.CellSizeArray();
579
580 // Set t_last to 0.0 to signify information read in
581 wdata.t_last = 0.0;
582
583 // indvec is complete upon initialization (all heights every proc)
584 amrex::Vector<int> indvec(wdata.hvec.size());
585 std::iota(indvec.begin(), indvec.end(), 0);
586 // Interpolate to MultiFabs (one level at a time)
587 interp_to_mfab::interp_eta_to_levelset_multifab(
588 wdata.n0_sp, wdata.n1_sp, wdata.dx0, wdata.dx1, wdata.xlo0,
589 wdata.xlo1, wdata.zsl, wdata.is_ocean, wdata.sp_eta_vec,
590 ow_levelset(level), problo, dx);
591 interp_to_mfab::interp_velocity_to_multifab(
592 wdata.n0_sp, wdata.n1_sp, wdata.dx0, wdata.dx1, wdata.xlo0,
593 wdata.xlo1, wdata.is_ocean, indvec, wdata.hvec, wdata.sp_u_vec,
594 wdata.sp_v_vec, wdata.sp_w_vec, ow_velocity(level), problo, dx);
595 // Zero velocity in pure air cells
596 postprocess_velocity_mfab_liquid(
597 ow_velocity(level), ow_levelset(level), dx);
598
599 // Populate flow fields according to intended forcing and init setup
600 const auto& ow_phi = ow_levelset(level).const_arrays();
601 const auto& ow_vel = ow_velocity(level).const_arrays();
602 const auto& vel = velocity(level).arrays();
603 const auto& phi_arrs = multiphase_mode
604 ? (*levelset)(level).arrays()
605 : amrex::MultiArray4<amrex::Real>();
606
607 const auto& w2a_phi = w2a_levelset(level).arrays();
608 const auto& w2a_vel = w2a_velocity(level).arrays();
609
610 const amrex::Real gen_length = wdata.gen_length;
611 const amrex::Real beach_length = wdata.beach_length;
612 const amrex::Real zero_sea_level = wdata.zsl;
613
614 const bool has_beach = wdata.has_beach && multiphase_mode;
615 const bool init_wave_field = wdata.init_wave_field || !multiphase_mode;
616
617 amrex::ParallelFor(
618 velocity(level), amrex::IntVect(3),
619 [=] AMREX_GPU_DEVICE(int nbx, int i, int j, int k) noexcept {
620 const amrex::Real x = problo[0] + (i + 0.5) * dx[0];
621 const amrex::Real z = problo[2] + (k + 0.5) * dx[2];
622
623 // Wave profile
624 const utils::WaveVec wave_sol{
625 ow_vel[nbx](i, j, k, 0), ow_vel[nbx](i, j, k, 1),
626 ow_vel[nbx](i, j, k, 2), ow_phi[nbx](i, j, k) + z};
627 // Quiescent profile
628 const utils::WaveVec quiescent{0.0, 0.0, 0.0, zero_sea_level};
629
630 // Specify initial state for each region of domain
631 const auto bulk = init_wave_field ? wave_sol : quiescent;
632 const auto outlet = has_beach ? quiescent : wave_sol;
633
634 const auto local_profile = utils::harmonize_profiles_1d(
635 x, problo[0], gen_length, probhi[0], beach_length, wave_sol,
636 bulk, outlet);
637
638 const amrex::Real phi = local_profile[3] - z;
639 const amrex::Real cell_length_2D =
640 std::sqrt(dx[0] * dx[0] + dx[2] * dx[2]);
641 if (phi + cell_length_2D >= 0) {
642 vel[nbx](i, j, k, 0) = local_profile[0];
643 vel[nbx](i, j, k, 1) = local_profile[1];
644 vel[nbx](i, j, k, 2) = local_profile[2];
645 }
646
647 if (multiphase_mode) {
648 phi_arrs[nbx](i, j, k) = phi;
649 }
650
651 // Default w2a values matter for where no updates happen
652 w2a_phi[nbx](i, j, k) = quiescent[3] - z;
653 w2a_vel[nbx](i, j, k, 0) = quiescent[0];
654 w2a_vel[nbx](i, j, k, 1) = quiescent[0];
655 w2a_vel[nbx](i, j, k, 2) = quiescent[0];
656 });
657 amrex::Gpu::streamSynchronize();
658#else
659 amrex::ignore_unused(data, level, geom, multiphase_mode);
660#endif
661 }

664template <>
665struct UpdateTargetFieldsOp<W2AWaves>
666{
667 // cppcheck-suppress constParameterReference
668 void operator()(W2AWaves::DataType& data, const amrex::Real time)
669 {
670
671#ifdef AMR_WIND_USE_W2A
672 auto& wdata = data.meta();
673 auto& sim = data.sim();
674
675 // Update ow fields every time
676 auto& ow_levelset = sim.repo().get_field("ow_levelset");
677 auto& ow_velocity = sim.repo().get_field("ow_velocity");
678 // Update HOS fields when necessary
679 auto& w2a_levelset = sim.repo().get_field("w2a_levelset");
680 auto& w2a_velocity = sim.repo().get_field("w2a_velocity");
681
682 // Proxy for multiphase mode of ocean waves
683 bool vof_exists = sim.repo().field_exists("vof");
684
685 auto nlevels = sim.repo().num_active_levels();
686 auto geom = sim.mesh().Geom();
687
688 // Get value for time interpolation
689 amrex::Real t_last = wdata.t_last;
690
691 // Check if new HOS data needs to be read
692 bool read_flag = false;
693 // Check if time indicates reading must take place
694 int new_ntime =
695 wdata.is_ocean
696 ? wdata.c_rmodes.time2step(time + wdata.t_winit, wdata.ntime)
697 : wdata.r_rmodes.time2step(time + wdata.t_winit, wdata.ntime);
698 int double_data = evaluate_read_resize(
699 wdata.ntime, read_flag, wdata.resize_flag, wdata.t, wdata.t_last,
700 new_ntime, wdata.t_winit, wdata.dt_modes, time);
701 // Check if stop time is exceeded, introduce offset to ntime
702 if (read_flag) {
703 // Need to only check when reading is occurring
704 if (wdata.ntime + wdata.n_offset > wdata.n_wstop) {
705 // If exceeding stop step, calculate new offset
706 wdata.n_offset =
707 update_offset_timestep(wdata.ntime, wdata.n_winit);
708 // Print warning to screen
709 amrex::Print()
710 << "WARNING (waves2amr_ops): available mode data exceeded, "
711 "resetting to beginning of mode data.\n";
712 }
713 }
714 // Resizing (assuming reading is taking place) must happen after regrid
715 if (wdata.regrid_occurred) {
716 // resize_flag remains true until resizing occurs, but
717 // regrid_occurred resets every timestep
718 wdata.resize_flag = true;
719 }
720
721 // Read HOS data if necessary based on time
722 if (read_flag) {
723
724 if (wdata.resize_flag) {
725 // Reset flag
726 wdata.resize_flag = false;
727 // Flags for indicating overlap, assume none at first
728 bool flag_z = false;
729 bool flag_xlo = false;
730 bool flag_xhi = false;
731 // Get heights for this processor, check overlap in z
732 flag_z =
733 (interp_to_mfab::get_local_height_indices(
734 wdata.indvec, wdata.hvec, ow_velocity.vec_ptrs(),
735 geom) == 1);
736 // No overlap from heights definitely means no interp
737
738 // Check lateral bounds (in x)
739 const int dir = 0;
740 flag_xlo =
741 (interp_to_mfab::check_lateral_overlap_lo(
742 wdata.gen_length, dir, ow_velocity.vec_ptrs(), geom) ==
743 1);
744 // No overlap with gen region means no interp, unless ...
745 if (wdata.has_outprofile) {
746 // ... if overlap exists here, needing interp
747 flag_xhi =
748 (interp_to_mfab::check_lateral_overlap_hi(
749 wdata.beach_length, dir, ow_velocity.vec_ptrs(),
750 geom) == 1);
751 }
752
753 // Wave generation zones are irrelevant for single-phase mode,
754 // indicated by nonexistence of vof
755 if (!vof_exists) {
756 flag_xlo = true;
757 flag_xhi = true;
758 }
759
760 if (flag_z && (flag_xlo || flag_xhi)) {
761 // Interpolation is needed
762 wdata.do_interp = true;
763 // Do resizing
764 wdata.sp_eta_vec.resize(
765 static_cast<size_t>(wdata.n0) *
766 static_cast<size_t>(wdata.n1),
767 0.0);
768 wdata.sp_u_vec.resize(
769 static_cast<size_t>(wdata.n0 * wdata.n1) *
770 wdata.indvec.size());
771 wdata.sp_v_vec.resize(
772 static_cast<size_t>(wdata.n0 * wdata.n1) *
773 wdata.indvec.size());
774 wdata.sp_w_vec.resize(
775 static_cast<size_t>(wdata.n0 * wdata.n1) *
776 wdata.indvec.size());
777 // Sizes will remain constant and need for interpolation
778 // will remain until a regrid occurs
779 } else {
780 // No overlapping with spatial data or no overlapping with
781 // relaxation zones, interpolation can be skipped
782 wdata.do_interp = false;
783 }
784 }
785 // Only perform reading where needed, communicate offset though
786 amrex::ParallelDescriptor::ReduceIntMax(wdata.n_offset);
787
788 // If double read is required, then copy older wave data to ow_
789 // fields and modify interpolation parameters to get things right
790 if (double_data == 1) {
791 if (wdata.do_interp) {
792 populate_fields_all_levels(
793 wdata, geom, ow_levelset, ow_velocity, -1);
794 }
795
796 // Average down to get fine information on coarse grid where
797 // possible (may be unnecessary)
798 for (int lev = nlevels - 1; lev > 0; --lev) {
799 amrex::average_down(
800 ow_velocity(lev), ow_velocity(lev - 1), 0,
801 AMREX_SPACEDIM, sim.mesh().refRatio(lev - 1));
802 amrex::average_down(
803 ow_levelset(lev), ow_levelset(lev - 1), 0, 1,
804 sim.mesh().refRatio(lev - 1));
805 }
806 // Fill patch to get correct ghost cells after average down
807 ow_velocity.fillpatch(sim.time().new_time());
808 ow_levelset.fillpatch(sim.time().new_time());
809
810 // Prior t_last (corresponding to ow fields)
811 t_last = (wdata.ntime - 1) * wdata.dt_modes;
812 } else if (double_data == 2) {
813 // Restarting simulation or taking a big step, new time at ntime
814 // Initialize ow fields to 0 for time interp, will be replaced
815 if (wdata.do_interp) {
816 ow_levelset.setVal(0.0, ow_levelset.num_grow()[0]);
817 ow_velocity.setVal(0.0, ow_levelset.num_grow()[0]);
818 }
819 // No modification needed for t_last, leads to interp factor = 1
820 }
821
822 // After possible prior read, now read data for this ntime
823 if (wdata.do_interp) {
824 populate_fields_all_levels(
825 wdata, geom, w2a_levelset, w2a_velocity);
826 }
827
828 // Average down to get fine information on coarse grid where
829 // possible and update ghost cells (may be unnecessary)
830 for (int lev = nlevels - 1; lev > 0; --lev) {
831 amrex::average_down(
832 w2a_velocity(lev), w2a_velocity(lev - 1), 0, AMREX_SPACEDIM,
833 sim.mesh().refRatio(lev - 1));
834 w2a_velocity(lev - 1).FillBoundary(geom[lev - 1].periodicity());
835 amrex::average_down(
836 w2a_levelset(lev), w2a_levelset(lev - 1), 0, 1,
837 sim.mesh().refRatio(lev - 1));
838 w2a_levelset(lev - 1).FillBoundary(geom[lev - 1].periodicity());
839 }
840 }
841
842 // Temporally interpolate at every timestep to get target solution
843 for (int lev = 0; lev < nlevels; ++lev) {
844 auto phi = ow_levelset(lev).arrays();
845 auto vel = ow_velocity(lev).arrays();
846 const auto W2A_phi = w2a_levelset(lev).const_arrays();
847 const auto W2A_vel = w2a_velocity(lev).const_arrays();
848
849 const amrex::Real W2A_t = wdata.t;
850 amrex::ParallelFor(
851 ow_levelset(lev), amrex::IntVect(3),
852 [=] AMREX_GPU_DEVICE(int nbx, int i, int j, int k) noexcept {
853 // Interpolate temporally every time
854 phi[nbx](i, j, k) +=
855 (W2A_phi[nbx](i, j, k) - phi[nbx](i, j, k)) *
856 (time - t_last) / (W2A_t - t_last + 1e-16);
857 vel[nbx](i, j, k, 0) +=
858 (W2A_vel[nbx](i, j, k, 0) - vel[nbx](i, j, k, 0)) *
859 (time - t_last) / (W2A_t - t_last + 1e-16);
860 vel[nbx](i, j, k, 1) +=
861 (W2A_vel[nbx](i, j, k, 1) - vel[nbx](i, j, k, 1)) *
862 (time - t_last) / (W2A_t - t_last + 1e-16);
863 vel[nbx](i, j, k, 2) +=
864 (W2A_vel[nbx](i, j, k, 2) - vel[nbx](i, j, k, 2)) *
865 (time - t_last) / (W2A_t - t_last + 1e-16);
866 });
867 }
868 amrex::Gpu::streamSynchronize();
869#else
870 amrex::ignore_unused(data, time);
871#endif
872 }
